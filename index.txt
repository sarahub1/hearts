    <h1>Welcome to the Falling Hearts Site!</h1>
    <button onclick="createHeart()">Click for Hearts</button>
    <br>
    <button id="cameraBtn" onclick="openCamera()">Open Camera</button>

    <!-- Video element to display the webcam feed -->
    <video id="camera" width="100%" height="auto" style="display: none;" autoplay></video>
    <div id="message" style="display: none; color: white; font-size: 40px; position: absolute; top: 50%; left: 50%; transform: translate(-50%, -50%);">
        GAHD DAMN YOU PRETTY
    </div>

    <script>
        // Function to create falling hearts
        function createHeart() {
            const heart = document.createElement('div');
            heart.classList.add('heart');
            heart.innerHTML = '❤️';
            heart.style.left = `${Math.random() * window.innerWidth}px`;
            document.body.appendChild(heart);
        }

        // Function to open the camera and show the video feed
        function openCamera() {
            const videoElement = document.getElementById('camera');
            const messageElement = document.getElementById('message');

            // Access the webcam
            navigator.mediaDevices.getUserMedia({ video: true })
                .then(stream => {
                    videoElement.srcObject = stream;
                    videoElement.style.display = 'block';
                    messageElement.style.display = 'block';

                    // After 3 seconds, hide the message (you can adjust timing)
                    setTimeout(() => {
                        messageElement.style.display = 'none';
                    }, 3000);
                })
                .catch(err => {
                    alert('Error accessing the camera: ' + err);
                });
        }
    </script>
